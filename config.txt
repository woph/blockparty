default.name=Default                   # name to display in the web interface
default.rpchost=127.0.0.1              # IP address of MultiChain node
default.rpcport=<yourPORT>             # see rpc-port from chain parameters

default.rpcpassword=<yourPasswort>     # password for RPC from multichain.conf
